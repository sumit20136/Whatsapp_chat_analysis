public class Sumit {
    private static Scanner sc = new Scanner(System.in);
    private static ArrayList<Point> all = new ArrayList<>();
    private static double low = 1e20;


    public static double dist(int x1, int y1, int x2, int y2) {
        return Math.sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
    }

    public static double findMin(int size) {
        for (int i = 0; i < size; i++) {
            for (int j = i + 1; j < size; j++) {
                for (int k = j + 1; k < size; k++) {
                    double perimeter = dist(all.get(i).getX(), all.get(i).getY(), all.get(j).getX(), all.get(j).getY()) + dist(all.get(i).getX(), all.get(i).getY(), all.get(k).getX(), all.get(k).getY()) + dist(all.get(j).getX(), all.get(j).getY(), all.get(k).getX(), all.get(k).getY());
                    if (perimeter < low) {
                        low = perimeter;
                    }
                    else {
                        continue;
                    }
                }
            }
        }
        return low;
    }

    public static void main(String args[]) throws IOException {

        int n = size;
        while (n > 0) {
            int x = sc.nextInt();
            int y = sc.nextInt();
            Point point = new Point(x, y);

Since, here the complexity for running the prims algorithm with only 3 such different edges, the answer comes out to be O(|E|+|V|), as the time complexity now does not have any role for priority queue to play.

        distance[source][i] = 0;

        PriorityQueue<AdjacencyListNode> priorityQueue = new PriorityQueue<>((v1, v2) -> v1.getWeight()-v2.getWeight());
